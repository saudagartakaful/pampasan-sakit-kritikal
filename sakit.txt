    <script>
        // Set current year in footer
        document.getElementById('currentYear').textContent = new Date().getFullYear();

        // Gemini API Integration
        const apiKey = ""; // API Key akan diuruskan oleh persekitaran Canvas
        const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;

        const askAiPlanButton = document.getElementById('askAiPlanButton');
        const aiPlanQuestionInput = document.getElementById('aiPlanQuestion');
        const aiPlanResponseDiv = document.getElementById('aiPlanResponse');
        const aiPlanResponseText = document.getElementById('aiPlanResponseText');
        const aiPlanLoadingSpinner = aiPlanResponseDiv.querySelector('.loading-spinner');

        // Fungsi untuk memanggil Gemini API
        async function callGeminiAPI(prompt, responseTextElement, loadingSpinnerElement, responseDivElement) {
            responseTextElement.textContent = '';
            loadingSpinnerElement.classList.remove('hidden');
            responseDivElement.classList.remove('hidden');

            try {
                const payload = {
                    contents: [{
                        role: "user",
                        parts: [{ text: prompt }]
                    }]
                };
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });

                if (!response.ok) {
                    throw new Error(`API request failed with status ${response.status}`);
                }

                const result = await response.json();
                
                if (result.candidates && result.candidates.length > 0 &&
                    result.candidates[0].content && result.candidates[0].content.parts &&
                    result.candidates[0].content.parts.length > 0) {
                    responseTextElement.textContent = result.candidates[0].content.parts[0].text;
                } else {
                    responseTextElement.textContent = "Maaf, saya tak dapat bagi respon buat masa ni. Cuba lagi nanti ye.";
                }

            } catch (error) {
                console.error("Error calling Gemini API:", error);
                responseTextElement.textContent = "Adoi, ada masalah sikit nak contact saya. Cuba check internet & try lagi.";
            } finally {
                loadingSpinnerElement.classList.add('hidden');
            }
        }

        // Event listener untuk "Tanya AI Pasal Plan"
        if (askAiPlanButton) {
            askAiPlanButton.addEventListener('click', () => {
                const userQuestion = aiPlanQuestionInput.value.trim();
                if (!userQuestion) {
                    aiPlanResponseDiv.classList.remove('hidden');
                    aiPlanResponseText.textContent = "Jangan lupa taip soalan dulu ye!";
                    return;
                }
                // Konteks yang lebih jelas untuk AI, seolah-olah "saya" (agen/pembantu maya) yang menjawab
                const planContext = "Anda adalah saya, seorang pembantu maya yang peramah untuk pelan pampasan sakit kritikal. Pelan ini menawarkan: Pampasan tunai RM100,000 jika disahkan sakit kritikal, pampasan peringkat awal RM10,000, ada nilai tunai, dan hibah RM100,000. Harga bermula RM89/bulan. Jawab soalan pengguna dengan ringkas (dalam 2-3 ayat), positif, dan dalam gaya bahasa Melayu santai (macam budak TikTok), fokus kepada bagaimana pelan ini membantu. Gunakan ganti nama 'saya' apabila merujuk kepada diri sendiri sebagai pembantu.";
                const fullPrompt = `${planContext}\n\nSoalan pengguna: ${userQuestion}\n\nJawapan saya (pastikan mesra dan mudah faham):`;
                
                callGeminiAPI(fullPrompt, aiPlanResponseText, aiPlanLoadingSpinner, aiPlanResponseDiv);
            });
        }
